         public int ExpMonth { get; set; }
         public int ExpYear { get; set; }

         {
             get
             {
                 if (ExpMonth < 10)
                     return "0" + ExpMonth.ToString() + ExpYear.ToString();
                 else
                     return ExpMonth.ToString() + ExpYear.ToString();
             }
         }
